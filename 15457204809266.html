微信小程序-网络请求封装与页面跳转

var app = getApp();
/**
 * POST请求，
 * URL：接口
 * postData：参数，json类型
 * doSuccess：成功的回调函数
 * doFail：失败的回调函数
 * 
 * 验证token的post请求
 * 
 */
function requestAuthPOST(url, postData, doSuccess, doFail) {
  console.log('-------------------请求开始-------------------')
  console.log("requestAuthPOST请求的url--->" + app.data.baseURL + url)
  console.log("请求的user_token--->" + getApp().data.user_token)
  console.log('请求参数如下：')
  console.log(postData)

  wx.request({
    url: app.data.baseURL + url,
    header: {
      'content-type': 'application/json',
      'Authorization': 'Bearer ' + getApp().data.user_token,
    },
    data: postData,
    method: 'POST',
    success: function(res) {
      console.log('请求success方法里返回的数据--->')
      console.log(res.data)
      console.log('-------------------请求结束-------------------')
      doSuccess(res.data);
    },
    fail: function(res) {
      console.log('请求fail方法里返回的数据--->')
      console.log(res)
      console.log('-------------------请求结束-------------------')
      doFail(res.data);
    },
  })
}

微信小程序点击方法传参与跳页传参

点击事件
bindtap  事件绑定不会阻止冒泡事件向上冒泡
catchtap  事件绑定可以阻止冒泡事件向上冒泡

点击事件的参数传递：
target：触发事件的源组件
currentTarget：事件绑定的当前事件

跳转页面

wx.navigateTo({
url: '../Temp/Temp?province=' + this.data.name + '&selectedindex=' + index + '&arrayIndex=' + JSON.stringify(this.data.arrayIndex),
})

小程序跳转方式

打开新页面,保有原来页面     调用 API wx.navigateTo

关闭当前页面，跳转到应用内的某个页面  调用 API wx.redirectTo

页面返回        调用 API wx.navigateBack

Tab 切换      调用 API wx.switchTab

重启动         调用 API wx.reLaunch

需要注意

1.navigateTo, redirectTo 只能打开非 tabBar 页面。
2.switchTab 只能打开 tabBar 页面。
3.reLaunch 可以打开任意页面。
4.页面底部的 tabBar 由页面决定，即只要是定义为 tabBar 的页面，底部都有 tabBar。
5.tabBar只能配置最少2个、最多5个 tab
6.调用页面路由带的参数可以在目标页面的onLoad中获取
