3..map 方法示例 接口返回的imageURL拼接根地址

var tempArray = '接口返回的数组'
that.setData({
    listArray: tempArray.map(function(item) {
        item.imageURL = getApp().data.baseURL + item.imageURL;
        //如果列表里没有子数组数据需要拼接根地址，就不用再次 .map 方法 直接return item
        item.child.map(function(childItem) {
          childItem.pic = getApp().data.baseURL + childItem.pic;
          return childItem;
        })
        return item;
    }),
})

8.倒计时

LastTimer: function() {

var that = this;
var totalSecond = 0;
//start_time服务端返回的团购开始时间
var start_time = new Date(that.data.start_time.replace(/-/g, "/"));
totalSecond = Date.parse(start_time) / 1000 - Date.parse(new Date()) / 1000;

    //清理掉倒计时
    clearInterval(that.data.interval_daojishi);
    //开始倒计时
    that.data.interval_daojishi = setInterval(function() {
      // 秒数
      var second = totalSecond;
      // 天数位
      var day = Math.floor(second / 3600 / 24);
      var dayStr = day.toString();
      if (dayStr.length == 1) dayStr = '0' + dayStr;
      // 小时位
      var hr = Math.floor((second - day * 3600 * 24) / 3600);
      var hrStr = hr.toString();
      if (hrStr.length == 1) hrStr = '0' + hrStr;
      // 分钟位
      var min = Math.floor((second - day * 3600 * 24 - hr * 3600) / 60);
      var minStr = min.toString();
      if (minStr.length == 1) minStr = '0' + minStr;
      // 秒位
      var sec = second - day * 3600 * 24 - hr * 3600 - min * 60;
      var secStr = sec.toString();
      if (secStr.length == 1) secStr = '0' + secStr;

      that.setData({
        countDownDay: dayStr,
        countDownHour: hrStr,
        countDownMinute: minStr,
        countDownSecond: secStr,
      });
      totalSecond--;

      if (totalSecond < 0) {
        //倒计时结束，清理掉倒计时
        clearInterval(that.data.interval_daojishi);
        that.setData({
          countDownDay: '00',
          countDownHour: '00',
          countDownMinute: '00',
          countDownSecond: '00',
        });
      }
    }.bind(this), 1000);
  },

9.定时请求调用

// *************定时请求的接口**************
//定时开始方法
  startSetInter: function () {
    var that = this;
    //将计时器赋值给setInter
    that.data.setInter = setInterval(
      function () {
       that.requestData();//被循环调用的方法 比如实时刷新数据
      }
      , 30000);
  },
  //定时结束方法
  endSetInter: function () {
    var that = this;
    //清除计时器  即清除setInter
    clearInterval(that.data.setInter)
  },

11.微信支付示例

//出门结算 微信支付
  OutDoorwxPay: function (payjine) {
    let that = this;
    wx.showLoading({
      title: '正在加载',
      mask: true,
    })
    // 登录
    wx.login({
      success: res => {
        // 发送 res.code 到后台换取 openId, sessionKey, unionId
        console.log(res)
        var dicdata = {
          id: that.data.wxpayId,
          qr_content: that.data.erweima,
          wxcode:res.code,
          total_fee: payjine
        }
        k_request.requestAuthPOST('create-wx-ord', dicdata, that.wxpay_doSuccess, that.wxpay_doFail);
      }
    })
  },
  wxpay_doSuccess: function (data) {
    wx.hideLoading();
    var that = this;
    if (data.result == 1) {
      that.wxpay(data.query)
    }else
    {
      wx.showToast({
        title: data.msg,
        icon:'none'
      })
    }
  },
  wxpay_doFail: function (data) {
    wx.hideLoading();
  },
  wxpay: function (data) {
    var that = this;
    wx.requestPayment({
      timeStamp: '' + data.timeStamp, //一定要传字符串 要不然会报错
      nonceStr: data.nonceStr,
      package: 'prepay_id=' + data.prepay_id,
      signType: 'MD5',
      appId: getApp().data.appid,
      paySign: data.paySign,

      success: function (pay) {
        console.log('支付返回的数据')
        console.log(pay)
        that.setData({
          showModal: false,
        })
        wx.showToast({
          title: '支付成功',
        })
        
        setTimeout(function () {
          //要延时执行的代码
          //返回到首页
          wx.navigateBack({
            delta: 1  // 返回上一级页面。
          });
        }, 1000) //延迟时间 这里是1秒

      },
      fail: function (pay) {
        console.log(pay)
       
        that.setData({
          showModal: false,
        })
        wx.showToast({
          title: '支付失败',
        })

        setTimeout(function () {
          //要延时执行的代码
          //返回到首页
          wx.navigateBack({
            delta: 1  // 返回上一级页面。
          });
        }, 1000) //延迟时间 这里是1秒
      }
    })
  },
