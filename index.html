Vue.js 监听属性 watch

通过watch来相应数据的变化，处理函数
目前 我们把它分为普通监听（也叫浅监听）和深度监听
  • 普通监听就是对变量直接进行监听
  • 深度监听就是对对象中的变量进行监听

例如：


<template>
  <div class="hello">
    <input class="colortext" text type="text" v-model="demo"><br>
    {{value}}<br>
    <input class="colortext" text type="text" v-model="obj.name"><br>
   {{value1}}<br>
   {{value2}}<br>
   {{value5}}<br>
    <input class="colortext" text type="text" v-model="num"><br>
   {{value3}}<br>
   {{value4}}<br>
   
  </div>
</template>

<script>
export default {
  name: 'HelloWorld',
  data () {
    return {
      msg: 'Welcome to Your Vue.js App',
      demo: '',
      value: '',
      num:1,
      obj:{
          name:'三儿',
          age:'21',
          sex:'女' 
        }
    }
  },
  watch:{
    num(val, oldVal){
    //普通的watch监听 
    console.log("num: "+val, oldVal);
     this.value3=val;
     this.value4=oldVal;
    },
    obj:{
    //深度监听，可监听到对象、数组的变化 此时val和oldVal值一样 如果需求要用到oldVal，须要用到 计算属性的方法computed，可得到上次oldVal的值
      handler(val, oldVal){
      console.log("obj.name: "+val.name, oldVal.name);
      this.value1=val.name;
      this.value2=oldVal.name;
      },
      deep:true
    },
    //深度监听时需用到oldVal时用，上次的值
    zhibianhua(newval,oldVal) {
      this.value5=oldVal;
    }
  },
  computed: {
    zhibianhua: function () {
      return this.obj.name
    }
  }
}
